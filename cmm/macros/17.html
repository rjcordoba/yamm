unsigned int pow(unsigned int base, unsigned int exp){
    unsigned int res;
    Y.obf = 1 - base;
    if(Z.obf){
      return 0;
    }
    Y.obf = 1 - exp;
    if(Z.obf){
      res = 1;
    }
    else{
      exp--;
      res = base * pow(base, exp);
    }
    return res;
}

unsigned int upow(unsigned int d, unsigned int v){
    unsigned int res;
    Y.obf = v % d;
    if(Z.obf){
      res = 0;
    }
    else{
      v = v / d;
      res = 1 + upow(d, v);
   }
   return res;
}
